const recordButton = document.getElementById('recordButton'); const feedbackDiv = document.getElementById('feedback'); const targetPhrase = "i am learning english"; const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition; if (SpeechRecognition) { const recognition = new SpeechRecognition(); recognition.lang = 'en-US'; recognition.interimResults = false; recordButton.onclick = () => { feedbackDiv.innerHTML = "Escuchando..."; recordButton.disabled = true; recognition.start(); }; recognition.onspeechend = () => { recognition.stop(); recordButton.disabled = false; }; recognition.onresult = (event) => { const result = event.results[0][0]; const transcript = result.transcript.toLowerCase().trim(); const confidence = result.confidence; if (transcript === targetPhrase) { feedbackDiv.innerHTML = `

¡Excelente! Tu pronunciación es perfecta.

Confianza del Reconocimiento: ${Math.round(confidence * 100)}%

`; speak("Excelente, tu pronunciación es perfecta.", "es-ES"); } else { const targetWords = targetPhrase.split(' '); const spokenWords = transcript.split(' '); let analysisHTML = ''; let correctWordsCount = 0; targetWords.forEach((targetWord, index) => { if (spokenWords[index] && spokenWords[index] === targetWord) { analysisHTML += `${targetWord} `; correctWordsCount++; } else { analysisHTML += `${targetWord} `; } }); const score = Math.round((correctWordsCount / targetWords.length) * 10); feedbackDiv.innerHTML = `

Inténtalo de nuevo.

Escuché: "${transcript}"

Análisis:

${analysisHTML}

Calificación: ${score} de 10

Confianza del Reconocimiento: ${Math.round(confidence * 100)}%

`; // --- ESTA ES LA PARTE MODIFICADA --- // Primero, habla en español. speak("La pronunciación correcta es...", "es-ES", () => { // Cuando termine de hablar en español, pronuncia la frase en inglés. speak("I am learning English", "en-US"); }); } }; recognition.onerror = (event) => { feedbackDiv.innerHTML = 'Hubo un error o no se detectó audio. Intenta de nuevo.'; recordButton.disabled = false; }; } else { feedbackDiv.innerHTML = "Lo siento, tu navegador no es compatible con esta función. Por favor, usa Google Chrome."; recordButton.disabled = true; } // --- Función para que la página HABLE --- function speak(text, lang, onEndCallback) { const utterance = new SpeechSynthesisUtterance(text); utterance.lang = lang; if (onEndCallback) { utterance.onend = onEndCallback; } window.speechSynthesis.speak(utterance); }
